	переменная1 = 0;
	переменная2 = 0;
	переменная3 = 0;
	вывод('введите строку, состоящую из скобок');
	ввод (СТРОКА);
		для всех элементов СТРОКи 	
			если элемент == '(' то
				переменная1 = переменная1 + 1
			иначе
				переменная2 = переменная2 + 1;

		Если переменная1 == переменная2 то
{
			для всех элементов СТРОКи
				если (элемент == '(') && (следующий элемент == ')') то
					переменная3 = переменная3 + 1
				иначе вывод ('Не валидно');
			
			если переменная3 == переменная1 то
				вывод ('Валидно'); 
}
		Иначе вывод ('Не валидно');
